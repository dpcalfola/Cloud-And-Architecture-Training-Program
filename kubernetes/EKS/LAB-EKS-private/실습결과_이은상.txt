# eksctl 명령어가 아닌 yaml 명세서를 통해 클러스터링 (dry-run)
# dry-run 결과는 나왔으나 실제 실행 실습은 하지 못했음


eksctl create cluster -f eksctl-cluster.yaml \
--dry-run | yh
